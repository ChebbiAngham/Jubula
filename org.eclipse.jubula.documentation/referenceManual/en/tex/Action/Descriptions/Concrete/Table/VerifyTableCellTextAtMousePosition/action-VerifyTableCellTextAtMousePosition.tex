% CAP description for Table --> Verify Text (Mouse Position)

\begin{itemize}
\item Use this action to check the text in a cell in a table based on the current position of the cursor.
\item This can be used in conjunction with the action  \bxcaption{select cell} with a click count of 0.  
\end{itemize}

\bxtipp{This action is unsupported for the HTML toolkit}
