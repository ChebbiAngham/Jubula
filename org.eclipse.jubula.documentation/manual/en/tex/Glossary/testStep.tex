%\index{Test Step}
%\gdhelpid{capContextId}{Test Steps}
\item[\gdstep]{
The smallest test object in \jb{}{}. A \gdstep belongs to a \gdcase, which is furthermore nested in a \gdsuite{}. A \gdstep  consists of 
\begin{itemize}
\item a \emph{{\bf C}omponent}
\item an \emph{{\bf A}ction} 
\item a \emph{{\bf P}arameter}
\end{itemize}
}
