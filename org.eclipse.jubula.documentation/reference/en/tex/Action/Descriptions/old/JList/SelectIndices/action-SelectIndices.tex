% CAP description for List --> Select Indices
\begin{itemize}
\item Use this action to select an item or items from the list. 
\item Enter the items you want to select using their indices.
\item \app{} realizes this action by:
\begin{itemize}
\item Pressing and holding \bxkey{CTRL}
\item Clicking on the indicated items
\item Releasing \bxkey{CTRL}
\end{itemize}
\end{itemize}
