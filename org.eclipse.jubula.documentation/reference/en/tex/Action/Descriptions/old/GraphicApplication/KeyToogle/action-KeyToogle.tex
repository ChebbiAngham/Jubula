% CAP description for Application --> Toggle Key
% 
\begin{itemize}
\item Use this action to turn the following keys on or off:
\begin{itemize}
\item Caps-lock
\item Num-lock
\item Scroll-lock
\end{itemize}
\item You specify which key to (de)activate and whether the function should be turned on or off.
\bxwarn{Due to current Java limitations, \gd does not support this action when testing under Linux. This means that any \gdstep containing this action will fail when testing under Linux.}
\end{itemize}
