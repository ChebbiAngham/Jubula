% CAP description for Tree --> Select by Indexpath --> Number of Clicks
\begin{itemize}
\item Enter the amount of clicks that should be performed on the node.
\item If you enter \bxshell{0}, the mouse pointer will just be moved over the specified node.
\item Entering \bxshell{1} will single-click on the node.
\item Entering \bxshell{2} will double-click on the node. 
\end{itemize}
