% CAP description for Table --> Move --> x-units
% 
\begin{itemize}
\item Use this parameter to tell \gd{} whether you want your values for the x-coordinate to be in pixels or percent.
\item  If you choose percent for  this parameter, and \bxshell{50} for the \bxname{x-position} parameter, the click will always be in the middle of the x-axis.
\end{itemize}
