% CAP description for Table --> Select Cell --> x-units
%
\begin{itemize}
\item Use this parameter to specify whether you want the value for your x-position to be in percent or in pixels.
\item  If you choose percent for  this parameter, and \bxshell{50} for the \bxname{x-position} parameter, the click will always be in the middle of the x-axis.
\end{itemize}
