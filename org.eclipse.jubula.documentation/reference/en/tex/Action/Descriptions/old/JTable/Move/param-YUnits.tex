% CAP description for Table --> Move --> y-units
%
\begin{itemize}
\item Use this parameter to tell \gd{} whether you want your values for the y-coordinate to be in pixels or percent.
\item If you choose percent for  this parameter, and \bxshell{50} for the \bxname{y-position} parameter, the click will always be in the middle of the y-axis.
\end{itemize}
