% CAP description for List --> Select Value(s)
\begin{itemize}
\item Use this action to select an item or items from the list. 
\item Enter the items you want to select using their names (values).
\item This is achieved by:
\begin{itemize}
\item Pressing and holding \bxkey{CTRL}
\item Clicking on the indicated items
\item Releasing \bxkey{CTRL}
\end{itemize}
\end{itemize}
\bxtipp{This action is unsupported for the HTML toolkit}
