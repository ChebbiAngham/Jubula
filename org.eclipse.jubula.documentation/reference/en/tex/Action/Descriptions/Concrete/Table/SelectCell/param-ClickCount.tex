% CAP description for Table --> Select Cell --> Number of Clicks

\begin{itemize}
\item Enter the amount of clicks that should be performed on the cell.
\item If you enter \bxshell{0}, the mouse pointer will just be moved over the specified cell.
\item Entering \bxshell{1} will single-click on the cell.
\item Entering \bxshell{2} will double-click on the cell. 
\end{itemize}
