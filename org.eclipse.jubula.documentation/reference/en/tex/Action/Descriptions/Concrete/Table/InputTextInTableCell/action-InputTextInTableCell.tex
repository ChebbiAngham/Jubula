% CAP description for Table --> Input Text (Specify Cell)
\begin{itemize}
\item Use this action to enter a text into a cell.
\item You specify the text to enter and the cell to enter it into. 

\app{}realizes this by:
\begin{itemize}
\item Clicking once on the cell to activate it
\item If the click does not activate the cell, \app{} double-clicks on the cell
\item Entering the text in the cell
\end{itemize}
\end{itemize}

\bxtipp{This action is unsupported for the HTML toolkit}
