% CAP description for Table --> Select Row
\begin{itemize}
\item Use this action to select the first cell in a row.
\item From this cell, you can navigate to other cells in the row, or to other rows.
\item \jb{} realizes this by clicking on the first element in the row. 
\end{itemize}

\bxtipp{This action is unsupported for the HTML toolkit}
